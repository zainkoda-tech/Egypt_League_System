import pandas as pd def calculate_league_table(csv_file): # الابن التاسع: التأكد من توافق المدخلات try: df = pd.read_csv(csv_file) teams = pd.unique(df[['Home', 'Away']].values.ravel('K')) stats = {team: {'لعب': 0, 'فوز': 0, 'تعادل': 0, 'هزيمة': 0, 'له': 0, 'عليه': 0, 'نقاط': 0} for team in teams} for _, row in df.iterrows(): home, away = row['Home'], row['Away'] # تحليل النتيجة (مثال: 4–0) score = str(row['Score']).replace('–', '-').split('-') if len(score) != 2: continue h_goals, a_goals = int(score[0]), int(score[1]) # تحديث الإحصائيات (الابن الخامس) stats[home]['لعب'] += 1 stats[away]['لعب'] += 1 stats[home]['له'] += h_goals stats[home]['عليه'] += a_goals stats[away]['له'] += a_goals stats[away]['عليه'] += h_goals if h_goals > a_goals: stats[home]['فوز'] += 1 ; stats[home]['نقاط'] += 3 stats[away]['هزيمة'] += 1 elif h_goals < a_goals: stats[away]['فوز'] += 1 ; stats[away]['نقاط'] += 3 stats[home]['هزيمة'] += 1 else: stats[home]['تعادل'] += 1 ; stats[home]['نقاط'] += 1 stats[away]['تعادل'] += 1 ; stats[away]['نقاط'] += 1 # ترتيب الجدول (النقاط ثم فارق الأهداف) table = pd.DataFrame.from_dict(stats, orient='index').sort_values(by=['نقاط', 'له'], ascending=False) return table except Exception as e: return f"عطب في البيانات: {e}" # تشغيل المحلل print("--- جدول ترتيب الدوري المصري 23/24 (تحليل نظام زين) ---") print(calculate_league_table('egypt_2024_results.csv'))
